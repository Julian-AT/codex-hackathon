       ```bash
       #!/usr/bin/env bash
       set -euo pipefail
       mkdir -p bench
       : "${MLX_MODEL:=unsloth/gemma-4-E4B-it-UD-MLX-4bit}"
       echo "== Bench start $(date -Iseconds) model=$MLX_MODEL ==" | tee bench/bench.log
       /usr/bin/time -l mlx_lm.lora \
         --model "$MLX_MODEL" \
         --train \
         --data ./data/smoke \
         --iters 50 \
         --batch-size 2 \
         --num-layers 16 \
         --max-seq-length 1024 \
         --grad-checkpoint \
         --steps-per-report 5 \
         --learning-rate 1e-5 \
         --adapter-path ./bench \
         2>&1 | tee -a bench/bench.log
       echo "== Bench end $(date -Iseconds) ==" | tee -a bench/bench.log
       # Cross-check MLX unified-memory peak (RESEARCH §Bench Command + Open Question #3)
       python3.12 -c "import mlx.core as mx; print('MLX_PEAK_BYTES=', mx.metal.get_peak_memory())" \
         2>&1 | tee -a bench/bench.log || echo "MLX_PEAK_BYTES=unavailable" | tee -a bench/bench.log
       ```
