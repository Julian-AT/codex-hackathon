       ```swift
       import Foundation
       import MLX
       import MLXLMCommon
       import MLXLLM
       import OSLog

       @MainActor
       final class AdapterSwap {
           private let log = Logger(subsystem: "com.hackathon.specialist", category: "adapter")
           static func documentsURL(_ name: String) -> URL {
               FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)
                   .first!.appendingPathComponent(name)
           }
           func swap(into container: ModelContainer) async throws -> TimeInterval {
               let url = Self.documentsURL("adapter.safetensors")
               let t0 = CFAbsoluteTimeGetCurrent()
               try await container.perform { ctx in
                   try LoRATrain.loadLoRAWeights(model: ctx.model, url: url)
               }
               let elapsed = CFAbsoluteTimeGetCurrent() - t0
               log.info("Adapter loaded in \(elapsed, privacy: .public) s from \(url.path, privacy: .public)")
               return elapsed
           }
       }
       ```

    3. Create `AdapterToolsLoader.swift` (depends on a `DynamicToolRegistrar` protocol that plan 05's `ToolRegistry` will conform to):
       ```swift
       import Foundation

       protocol DynamicToolRegistrar {
           func registerTool(name: String, jsBody: String) async
       }

       struct DynamicToolSpec: Decodable {
           let name: String
           let description: String
           let schema: [String: JSONValueBox]
           let jsBody: String
           let requiresNetwork: Bool
       }

       // Minimal JSON holder so the schema blob survives Decodable without forcing a full schema parser.
       struct JSONValueBox: Decodable {
           let raw: Data
           init(from decoder: Decoder) throws {
               let c = try decoder.singleValueContainer()
               if let d = try? c.decode(JSONValueBox.Raw.self) { self.raw = d.raw } else { self.raw = Data() }
           }
           struct Raw: Decodable { let raw: Data
               init(from d: Decoder) throws {
                   let c = try d.singleValueContainer()
                   self.raw = try JSONSerialization.data(withJSONObject: try c.decode(AnyDecodable.self).value)
               }
           }
       }
       // (Implementer: if the JSONValueBox dance is more fragile than it's worth, substitute
       //  `let schema: Data` and capture the raw JSON substring — Phase 1 doesn't read schema,
       //  it just keeps it in-memory for Phase 3's sandbox validator.)

       final class AdapterToolsLoader {
           func load(into registrar: DynamicToolRegistrar) async throws -> Int {
               let url = AdapterSwap.documentsURL("adapter-tools.json")
               let data = try Data(contentsOf: url)
               let specs = try JSONDecoder().decode([DynamicToolSpec].self, from: data)
               for spec in specs {
                   await registrar.registerTool(name: spec.name, jsBody: spec.jsBody)
               }
               return specs.count
           }
       }
       ```
