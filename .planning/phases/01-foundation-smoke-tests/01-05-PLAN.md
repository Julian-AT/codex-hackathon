       ```swift
       import Foundation
       import JavaScriptCore

       actor ToolRegistry: DynamicToolRegistrar {
           private let ctx: JSContext
           private var specsByName: [String: (requiresNetwork: Bool, schemaRaw: String)] = [:]
           private let onlineRef: () -> Bool   // reads monitor live (P4)

           init(onlineRef: @escaping () -> Bool) {
               self.ctx = JSContext()!
               self.onlineRef = onlineRef
               installBridges()
           }

           private func installBridges() {
               // console.log
               let consoleLog: @convention(block) (String) -> Void = { msg in
                   NSLog("[JS] %@", msg)
               }
               let console = JSValue(newObjectIn: ctx)!
               console.setObject(consoleLog, forKeyedSubscript: "log" as NSString)
               ctx.setObject(console, forKeyedSubscript: "console" as NSString)

               // nativeFetch(url, cb)
               // Synchronous URLSession.shared.data(from:) via a semaphore inside a Task; JS sees a promise-like
               // via a callback-based shim. Phase 1 only needs this to exist — echoArgs does not use it.
               let nativeFetch: @convention(block) (String, JSValue) -> Void = { [weak self] urlStr, cb in
                   guard let _ = self, let url = URL(string: urlStr) else {
                       _ = cb.call(withArguments: [["error":"bad_url"] as Any])
                       return
                   }
                   Task.detached {
                       do {
                           let (data, _) = try await URLSession.shared.data(from: url)
                           let s = String(data: data, encoding: .utf8) ?? ""
                           await MainActor.run { _ = cb.call(withArguments: [["ok":true,"body":s] as Any]) }
                       } catch {
                           await MainActor.run { _ = cb.call(withArguments: [["error": "\(error)"] as Any]) }
                       }
                   }
               }
               ctx.setObject(nativeFetch, forKeyedSubscript: "nativeFetch" as NSString)
           }

           func registerTool(name: String, jsBody: String) async {
               // Phase 1: default requiresNetwork=false (echoArgs). AdapterToolsLoader can call
               // registerToolFull to carry requiresNetwork through.
               ctx.evaluateScript("globalThis[\"\(name)\"] = function(args) { \(jsBody) };")
               specsByName[name] = (false, "")
           }

           func registerToolFull(name: String, jsBody: String, requiresNetwork: Bool) async {
               ctx.evaluateScript("globalThis[\"\(name)\"] = function(args) { \(jsBody) };")
               specsByName[name] = (requiresNetwork, "")
           }

           func invoke(name: String, argsJSON: String) async -> String {
               guard let spec = specsByName[name] else {
                   return #"{"error":"unknown_tool","name":"\#(name)"}"#
               }
               if spec.requiresNetwork && !onlineRef() {
                   // PRD §9.5 structured refusal
                   return #"{"error":"This tool requires network. Device is offline."}"#
               }
               guard let fn = ctx.objectForKeyedSubscript(name) else {
                   return #"{"error":"tool_not_installed"}"#
               }
               // Parse args JSON via JS Object literal (boundary: strings in/out)
               let argsVal = ctx.evaluateScript("(\(argsJSON))")
               let result = fn.call(withArguments: [argsVal as Any])
               // Stringify result
               let stringifier = ctx.evaluateScript("(function(v){return JSON.stringify(v);})")!
               let s = stringifier.call(withArguments: [result as Any])
               return s?.toString() ?? "{}"
           }
       }
       ```

       **Design notes baked in:**
       - `JSContext` is not Sendable → actor isolates it (RESEARCH §JSContext Bridge gotcha).
       - Single JSContext per registry; app lifetime (PRD §16 R8 OOM gate).
       - JSON boundary: strings in, strings out (never pass `[String:Any]` across JSValue).
       - `onlineRef` is a closure that reads `OnlineMonitor.isOnline` live per-call (P4 avoidance).

    2. Create `GemmaToolParser.swift`:
       ```swift
       import Foundation

       /// Intercepts `<|tool_call|>{...}<|tool_response|>` pairs in a decoded text stream.
       /// When a tool_call block is complete, dispatches via ToolRegistry and yields the
       /// `<|tool_response|>{json}<|end_of_turn|>` string back to the caller, which is
       /// responsible for re-feeding those tokens into the generation loop.
       actor GemmaToolParser {
           private let registry: ToolRegistry
           private var buffer: String = ""
           // PRD §8.2 literal; RESEARCH P5: verify empirically at H2:00 by logging raw stream.
           private let openTag = "<|tool_call|>"
           private let closeTag = "<|tool_response|>"
           private let endOfTurn = "<|end_of_turn|>"

           init(registry: ToolRegistry) { self.registry = registry }

           /// Feed a decoded text chunk. Returns:
           ///   - `.passthrough(text)` — forward to UI as-is.
           ///   - `.toolResponse(injectedString)` — caller must re-feed into generation state.
           enum Output { case passthrough(String); case toolResponse(String) }

           func feed(_ chunk: String) async -> [Output] {
               buffer += chunk
               var outputs: [Output] = []
               while let openRange = buffer.range(of: openTag) {
                   // Emit everything before the open tag as passthrough
                   let before = String(buffer[..<openRange.lowerBound])
                   if !before.isEmpty { outputs.append(.passthrough(before)) }
                   buffer.removeSubrange(..<openRange.upperBound)
                   // Wait for close tag (it may not have arrived yet)
                   guard let closeRange = buffer.range(of: closeTag) else {
                       // Re-prepend open tag back — we'll re-scan on next feed
                       buffer = openTag + buffer
                       return outputs
                   }
                   let jsonStr = String(buffer[..<closeRange.lowerBound])
                   buffer.removeSubrange(..<closeRange.upperBound)
                   // Dispatch
                   let injected = await dispatch(jsonStr: jsonStr)
                   outputs.append(.toolResponse(injected))
               }
               // Any remaining buffer is a partial passthrough only if it can't contain open tag
               if !buffer.contains("<|") {
                   outputs.append(.passthrough(buffer))
                   buffer = ""
               }
               return outputs
           }

           private func dispatch(jsonStr: String) async -> String {
               // JSON shape: {"name":"toolX","arguments":{...}}
               struct Call: Decodable { let name: String; let arguments: [String: JSONValueAny] }
               struct JSONValueAny: Decodable {
                   let raw: String
                   init(from d: Decoder) throws {
                       let c = try d.singleValueContainer()
                       if let s = try? c.decode(String.self) { self.raw = "\"\(s)\"" }
                       else if let n = try? c.decode(Double.self) { self.raw = "\(n)" }
                       else if let b = try? c.decode(Bool.self) { self.raw = "\(b)" }
                       else { self.raw = "null" }
                   }
               }
               guard let data = jsonStr.data(using: .utf8),
                     let call = try? JSONDecoder().decode(Call.self, from: data) else {
                   return "\(closeTag){\"error\":\"parse_failed\"}\(endOfTurn)"
               }
               // Reconstruct args JSON for ToolRegistry.invoke (expects a JSON object literal string)
               let argsJSON = "{" + call.arguments.map { "\"\($0.key)\":\($0.value.raw)" }.joined(separator: ",") + "}"
               let result = await registry.invoke(name: call.name, argsJSON: argsJSON)
               return "\(closeTag)\(result)\(endOfTurn)"
           }
       }
       ```
