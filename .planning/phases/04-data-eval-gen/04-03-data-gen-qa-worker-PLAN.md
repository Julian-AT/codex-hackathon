- MUST use `createAnthropic({ baseURL: 'https://api.anthropic.com' })` — the shell exports `ANTHROPIC_BASE_URL=http://localhost:4141` for Claude Code proxy; AI SDK routes must bypass this.
- `generateObject` from `'ai'` with `schema` param (AI SDK v6 structured output).
- `p-limit` from `'p-limit'` (ESM default import) — `const limit = pLimit(15);`
- Sentry: `import * as Sentry from '@sentry/nextjs';` + `Sentry.startSpan(...)`.

Prompt design (Genstruct x PersonaHub):
- System prompt: persona voice + "You are generating training data for a Supabase specialist model. Ground every answer in the provided documentation. If the question involves a tool, include a tool_call with correctly-formed arguments."
- User prompt: difficulty-framed question generation instruction + chunk text as [CONTEXT]. For 'easy': single fact lookup. For 'medium': synthesis across the provided chunks. For 'hard': multi-hop reasoning or schema-aware tool argument construction.
- Response schema (Zod): `{ question: string, answer: string, toolCalls?: Array<{ name: string, arguments: Record<string,unknown> }>, reasoning?: string }`.
- Post-process: convert Zod output into `TrainingExample.messages` array.

Rate limit respect:
- `p-limit(15)` wraps every `generateObject` call. PRD §7.1 says fan-out under p-limit(15) to respect rate limits.
- On 429: catch, exponential backoff (2^attempt * 1000 ms, max 3 retries), then re-enqueue.

       ```ts
       import { generateObject } from 'ai';
       import { createAnthropic } from '@ai-sdk/anthropic';
       import * as Sentry from '@sentry/nextjs';
       import pLimit from 'p-limit';
       import { readFileSync } from 'node:fs';
       import path from 'node:path';
       import type { Chunk, DynamicToolSpec } from '../discovery/types.js';
       import type { TrainingExample, ChatMessage, ToolCall, DataGenMeta } from './types.js';
       import { PERSONAS, DIFFICULTIES, samplePersona, sampleDifficulty, makeRng } from './personas.js';
       import { validateToolCall } from './schema-gate.js';
       import { QA_RESPONSE_SCHEMA, buildQASystemPrompt, buildQAUserPrompt } from './qa-prompts.js';

       // CRITICAL: pin baseURL to bypass shell-shadowed ANTHROPIC_BASE_URL
       const anthropicProvider = createAnthropic({
         baseURL: 'https://api.anthropic.com',
       });
       const MODEL = anthropicProvider('claude-opus-4-7');

       export interface QABatchOptions {
         trainChunks: Chunk[];
         tools: DynamicToolSpec[];
         count?: number;        // default 500
         concurrency?: number;  // default 15
         seed?: string;         // default 'qa-gen-v1'
         maxRetries?: number;   // default 2
         onProgress?: (done: number, total: number) => void;
       }

       export interface QABatchResult {
         examples: TrainingExample[];
         meta: DataGenMeta[];
         rejected: number;
       }

       export async function generateQABatch(opts: QABatchOptions): Promise<QABatchResult> {
         const {
           trainChunks, tools, count = 500, concurrency = 15,
           seed = 'qa-gen-v1', maxRetries = 2, onProgress,
         } = opts;
         const limit = pLimit(concurrency);
         const rng = makeRng(seed);
         const examples: TrainingExample[] = [];
         const meta: DataGenMeta[] = [];
         let rejected = 0;
         let done = 0;

         // Pre-compute assignments: persona x difficulty x chunk window
         const assignments = Array.from({ length: count }, (_, i) => {
           const persona = samplePersona(rng);
           const difficulty = sampleDifficulty(rng);
           // Select 1-3 chunks depending on difficulty
           const numChunks = difficulty === 'easy' ? 1 : difficulty === 'medium' ? 2 : 3;
           const startIdx = Math.floor(rng() * trainChunks.length);
           const selectedChunks: Chunk[] = [];
           for (let j = 0; j < numChunks; j++) {
             selectedChunks.push(trainChunks[(startIdx + j) % trainChunks.length]);
           }
           return { index: i, persona, difficulty, chunks: selectedChunks };
         });

         const generateOne = async (
           assignment: typeof assignments[0],
           attempt = 0,
           negFeedback?: string,
         ): Promise<void> => {
           const { persona, difficulty, chunks } = assignment;
           const chunkIds = chunks.map(c => c.id);

           try {
             const result = await Sentry.startSpan(
               { op: 'ai.agent', name: 'data-gen-qa' },
               async (span) => {
                 span.setAttribute('persona', persona.id);
                 span.setAttribute('difficulty', difficulty);
                 span.setAttribute('chunkIds', chunkIds.join(','));
                 span.setAttribute('attempt', attempt);

                 const systemPrompt = buildQASystemPrompt(persona, tools);
                 let userPrompt = buildQAUserPrompt(difficulty, chunks);
                 if (negFeedback) {
                   userPrompt += `\n\n[PREVIOUS ATTEMPT REJECTED: ${negFeedback}. Fix the issue.]`;
                 }

                 const { object } = await generateObject({
                   model: MODEL,
                   schema: QA_RESPONSE_SCHEMA,
                   system: systemPrompt,
                   prompt: userPrompt,
                   temperature: 0.7,
                   experimental_telemetry: { isEnabled: true, functionId: 'data-gen-qa' },
                 });
                 return object;
               },
             );

             // Schema-gate: validate any tool_calls
             if (result.toolCalls && result.toolCalls.length > 0) {
               for (const tc of result.toolCalls) {
                 const validation = validateToolCall(tc.name, tc.arguments);
                 if (!validation.valid) {
                   if (attempt < maxRetries) {
                     return generateOne(
                       assignment,
                       attempt + 1,
                       `Tool call to '${tc.name}' failed schema validation: ${validation.errors?.join('; ')}`,
                     );
                   }
                   rejected++;
                   return; // DAT-03: reject, never patch
                 }
               }
             }

             // Convert to TrainingExample
             const messages: ChatMessage[] = [
               { role: 'system', content: `You are a Supabase specialist assistant. ${persona.voice}` },
               { role: 'user', content: result.question },
             ];

             if (result.toolCalls && result.toolCalls.length > 0) {
               const toolCallObjs: ToolCall[] = result.toolCalls.map((tc, idx) => ({
                 id: `call_${assignment.index}_${idx}`,
                 type: 'function' as const,
                 function: { name: tc.name, arguments: JSON.stringify(tc.arguments) },
               }));
               messages.push({
                 role: 'assistant',
                 content: result.reasoning ?? '',
                 tool_calls: toolCallObjs,
               });
               // Add tool responses (simulated from adapter-tools.json jsBody execution could go here,
               // but for QA the answer IS the response — keep it simple)
               for (const tc of toolCallObjs) {
                 messages.push({
                   role: 'tool',
                   content: JSON.stringify({ result: 'See answer below' }),
                   tool_call_id: tc.id,
                   name: tc.function.name,
                 });
               }
               messages.push({ role: 'assistant', content: result.answer });
             } else {
               messages.push({ role: 'assistant', content: result.answer });
             }

             examples.push({ messages, tools });
             meta.push({
               persona: persona.id,
               difficulty,
               sourceChunks: chunkIds,
               generator: 'opus-4-7',
             });
           } catch (err) {
             const status = (err as { statusCode?: number })?.statusCode;
             if (status === 429 && attempt < maxRetries) {
               await new Promise(r => setTimeout(r, 2 ** attempt * 1000));
               return generateOne(assignment, attempt + 1);
             }
             rejected++;
           } finally {
             done++;
             onProgress?.(done, count);
           }
         };

         await Promise.all(
           assignments.map((a) => limit(() => generateOne(a))),
         );

         return { examples, meta, rejected };
       }
       ```
